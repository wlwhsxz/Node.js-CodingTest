```
// 시간초과

const input = require("fs")
  .readFileSync("dev/stdin")
  .toString()
  .trim()
  .split(" ")
  .map((v) => +v);

const A = input[0];
const B = input[1];
const C = input[2];

let bool = true;
let count = 1;

while (bool) {
  if (B >= C) {
    count = -1;
    bool = false;
  } else if (A + B * count < C * count) {
    bool = false;
  } else {
    count++;
  }
}

console.log(count);
```
